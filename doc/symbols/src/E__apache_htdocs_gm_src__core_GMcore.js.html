1 /**
  2  * @author <a href="mailto:designsor@gmail.com" target="_blank">Fuqiang[designsor@gmail.com]</a>
  3  * @version 20110726
  4  * @fileoverview Goodmedia前端js库core文件，负责创建GM对象和构建命名空间 基于jquery1.6.2
  5  */
  6 (function(W, doc, $) {
  7 	/**
  8 	 * @namespace
  9 	 * @name GM
 10 	 * @description 全局的GM对象
 11 	 */
 12 
 13 	/**
 14 	 * @namespace
 15 	 * @name jQuery
 16 	 * @description 这里只对私自扩展的jQuery进行doc描述
 17 	 */
 18 
 19 	if (!W.GM) var GM = {};
 20 	/**
 21 	 * @namespace
 22 	 * @description 与业务紧密相关的-挂件
 23 	 */
 24 	GM.widget = {};
 25 	/**
 26 	 * @namespace
 27 	 * @description 与业务无关的比如overlay组件,最后扩展到jquery上，使用jquery的扩展机制进行最后的封装
 28 	 */
 29 	GM.tools = {};
 30 	/**
 31 	 * @namespace
 32 	 * @description 独立项目或者应用
 33 	 */
 34 	GM.apps = {};
 35 	/**
 36 	 * @constant
 37 	 * @description 根据href里德debug关键字确定是否为debug模式
 38 	 */
 39 	GM.debug = W.GMDEBUG.aver;
 40 	/**
 41 	 * @constant
 42 	 * @description 获取当前js的根目录
 43 	 */
 44 	GM.host = function() {
 45 		var scripts = doc.getElementsByTagName('script'),
 46 		i,
 47 		base;
 48 		for (i = 0; i < scripts.length; i++) {
 49 			var src = scripts[i].src,
 50 			namereg = /(GM-min|GM).js/;
 51 			if (namereg.test(src)) {
 52 				base = src.slice(0, src.lastIndexOf('/') + 1);
 53 			}
 54 		}
 55 		return base;
 56 	} ();
 57 	/**
 58 	 * @static
 59 	 * @description 转换到本地非压缩路径
 60 	 * @function
 61 	 * @param {string} uri
 62 	 * @returns {string} 处理过的路径
 63 	 */
 64 	GM.locality = function(uri) {
 65 		return uri.replace(/bulid/g, 'src').replace(/\-min/g, '');
 66 	};
 67 	/**
 68 	 * @constant
 69 	 * @description widget的根目录
 70 	 */
 71 	GM.widget.host = GM.host + 'widget/';
 72 	/**
 73 	 * @constant
 74 	 * @description apps的根目录
 75 	 */
 76 	GM.apps.host = GM.host + 'apps/';
 77 
 78 	//debug模式下处理路径
 79 	if (GM.debug) {
 80 		GM.host = W.GMDEBUG.host + 'bulid/';
 81 		GM.widget.host = GM.locality(GM.host) + 'widget/';
 82 		GM.apps.host = GM.locality(GM.host) + 'apps/';
 83 	}
 84 	/**
 85 	 * @static
 86 	 * @description 额外加载项目文件 - 项目文件目前依赖关系依靠ant维护
 87 	 * @function
 88 	 * @param {string} appname
 89 	 * @param {function} callback
 90 	 */
 91 
 92 	GM.apps.usemap = {};
 93 
 94 	GM.apps.require = function(appname, callback) {
 95 		if (!GM['@apps' + appname]) {
 96 			GM['@apps' + appname] = {};
 97 			GM['@apps' + appname].isLoading = true;
 98 		  var appuri = GM.host + 'apps/' + appname + '/' + appname + '-min.js';
 99 		  if (GM.debug) appuri = GM.locality(appuri);
100 			$(function() {
101 				$.getScript(appuri, function() {
102 					GM.apps.usemap[appname] = {
103 						uri: appuri
104 					};
105 					GM['@apps' + appname].Loaded = true;
106 					callback(GM.apps[appname]['exports']);
107 				});
108 			});
109 		} else {
110 			var T = setInterval(function() {
111 				if (GM['@apps' + appname] && GM['@apps' + appname].Loaded) {
112 					clearInterval(T);
113 					callback(GM.apps[appname]['exports']);
114 				}
115 			},
116 			500);
117 		}
118 	};
119 	/**
120 	 * @static
121 	 * @description 把用过的widget储存，再次use则不再调用
122 	 * @private
123 	 */
124 	GM.widget.usemap = {};
125 	/**
126 	 * @static
127 	 * @description 加载widget的方法
128 	 * @function
129 	 * @param {String} widget
130 	 * @param {function} callback
131 	 */
132 
133 	GM.widget.use = function(widget, callback) {
134 		if (!GM['@widget' + widget]) {
135 			GM['@widget' + widget] = {};
136 			GM['@widget' + widget].isLoading = true;
137 			var widgeturi = GM.host + 'widget/' + widget + '/' + widget + '-min.js';
138 			if (GM.debug) widgeturi = GM.locality(widgeturi);
139 			$(function() {
140 				$.getScript(widgeturi, function() {
141 					GM.widget.usemap[widget] = {
142 						uri: widgeturi
143 					};
144 					GM['@widget' + widget].Loaded = true;
145 					callback(GM.widget);
146 				});
147 			});
148 		} else {
149 			var T = setInterval(function() {
150 				if (GM['@widget' + widget] && GM['@widget' + widget].Loaded) {
151 					clearInterval(T);
152 					callback(GM.widget);
153 				}
154 			},
155 			500);
156 		}
157 	};
158 
159 	//debug 模式下开启debug.js并且初始化debug面板
160 	if (GM.debug) {
161 		GM.widget.use('debug', function(widget) {
162 			widget.debug.init();
163 		});
164 	}
165 
166 	GM.apps.require('idongmi', function(exports) {
167 		exports.init();
168 		exports.backtop();
169 	});
170 
171 	$(function() {
172 		if ($('#J_MuenbarV2').length > 0) {
173 			GM.apps.require('cmsmuenbarv2', function(exports) {
174 				exports.init();
175 			});
176 		}
177 	});
178 
179 	W.GM = GM;
180 })(window, document, jQuery);
