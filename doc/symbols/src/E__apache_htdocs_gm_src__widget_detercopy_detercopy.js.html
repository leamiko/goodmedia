  6  * IE下如果不全部屏蔽功能，则可以通过ctrl+a或者从其他节点复制过来，造成影响。这里封装为IE下强制屏蔽复制，什么都不可以选，不可以复制。</p>

  8 (function(W,doc,G){
  9 	
 10 	var detercopy=function(selector){
 11 		$(function(){
 12 		
 13 			if($(selector).length==0) return;
 14 			
 15 			if(document.attachEvent){
 16 				
 17 				var falseFn=function(){return false;};
 18 				
 19 				doc.body.attachEvent("onselectstart",falseFn);
 20 				
 21 				doc.body.attachEvent('oncontextmenu',falseFn);
 22 				
 23 				doc.body.attachEvent('oncopy',falseFn);
 24 				
 25 				doc.body.attachEvent('oncut',falseFn);
 26 				
 27 				$(document).keydown(function(){
 28 					if(event.ctrlKey && event.keyCode==67){
 29 						event.keyCode=0;
 30 						event.returnValue=false;
 31 					}
 32 				});
 33 				
 34 				if(doc.selection){
 35 					doc.selection.empty();
 36 					doc.unselectable = "on";
 37    					doc.documentElement.style['userSelect'] = "none";
 38 				}
 39 				
 40 			}else{
 41 				$(selector).each(function(index,node){
 42 					node.style.cssText+="-moz-user-select:none;-webkit-user-select:none;";
 43 				});
 44 			}
 45 		});
 46 	}
 47 	
 48 	if(G && G.widget){
 49 		G.widget.detercopy=detercopy;
 50 		/**
 51 		 * 主动触发一次,禁止复制的钩子为J_NoCopy
 52 		 */
 53 		G.widget.detercopy('.J_NoCopy');
 54 	} 
 55 	
 56 })(window,document,GM);
