127 		createPag:function(count,max,region){
128 			var that=this,cg=that.config,page=Math.ceil(count/max);
129 			
130 			if(count<=max) return ''; //不用分页，返回空
131 				
132 			var	pag='<div class="pcomment_page" id="'+cg.pagwrap.slice(1)+'"><div class="fanye">';
133 				
134 			if(count>max && cg.current!=1) pag+='<a class="a1" id="'+cg.pagback.slice(1)+'" href="javascript:void(0);">上一页</a>';
135 			
136 			var i=(cg.current-1==0)?1:cg.current-1,
137 				region=parseInt(cg.current)+parseInt(region-1);
138 			
139 			if(i!=1) pag+='<span class="mor">...</span>';
140 			
141 			for(;i<=region;i++){
142 				if(i>page) break;
143 				if(i==cg.current){
144 					pag+='<span>'+i+'</span>';
145 				}else{
146 					pag+='<a href="javascript:void(0);" class="'+cg.pag.slice(1)+'">'+i+'</a>';
147 				}
148 			}
149 			if(i<=page) pag+='<span class="mor">...</span>';
150   					
151   			if(count>max && cg.current!=page) pag += '<a class="a1" id="'+cg.pagfowd.slice(1)+'" href="javascript:void(0);">下一页</a>';
152   			
153   			return pag+='</div></div>';

347 		getDmName:function(selector){

349 			
350 			$(selector).each(function(){
351 				names.push($(this).attr('data-name'));
352 			});
353

357 				url:'/api/reply/replyAjax.jsp',
358 				data:{act:'guids',type:cg.type,pid:1,unames:names.join(' ')},
359 				type:'GET',
360 				success:function(result){
361 					cg.deadlock=true;
362 					var result=$.trim(result),
363 						uids=result.split(',');
364 					for(var i=0;i<uids.length;i++){
365 						if(uids[i]==0){
366 							$(selector).eq(i).attr({
367 								'href':'javascript:void(0)',
368 								'target':'',
369 								'title':'此用户不存在'
370 							}).addClass('Noname');
371 						}else{
372 							$(selector).eq(i).attr('href','/u/'+uids[i]);
373 						}
374 					}	
375 				},
376 				error:function(){
377 					cg.deadlock=true;
378 					$(selector).each(function(){
379 						$(this).attr({
380 								'href':'javascript:void(0)',
381 								'target':'',
