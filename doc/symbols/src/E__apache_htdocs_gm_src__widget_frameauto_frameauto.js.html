  6 (function(W,G,$) {

  8 	function iframeAutoFit() {
  9 		try {
 10 			if (self !== top) {
 11 				var a = parent.document.getElementsByTagName("IFRAME");
 12 				for (var i = 0; i < a.length; i++) {
 13 					if (a[i].contentWindow == window) {
 14 						var h1 = 0,
 15 						h2 = 0,

 23 						if (document.all) {
 24 							h += 4;
 25 						}
 26 						if (window.opera) {
 27 							h += 1;
 28 						}
 29 						a[i].style.height = h + "px";
 30 						//iframe h
 31 						a[i].parentNode.style.height = h + "px";
 32 					}
 33 				}
 34       }
 35     }
 36     catch(ex) {}
 37 	}
 38   
 39   if(G && G.widget){
 40     G.widget.frameauto=iframeAutoFit;
 41   }
