1 /**
  2  * @author <a href="mailto:designsor@gmail.com" target="_blank">Fuqiang[designsor@gmail.com]</a>
  3  * @version 20110725
  4  * @fileoverview 为GM对象添加widget空间下的countdown组件,我说倒计时
  5  */
  6 (function(W,G){
  7 	/**
  8 	 * @memberOf GM.widget
  9 	 * @class
 10 	 * @description 
 11 	 */
 12 	var saycountdown=function(cg){
 13 		
 14 		var fn={
 15 			init:function(){
 16 				return new fn._bind(cg);	
 17 			},
 18 			//绑定事件，创建构造器
 19 			_bind:function(cg){
 20 				var _cg={
 21 					main:'',
 22 					wrap:'',
 23 					errorCls:'',
 24 					holdTarget:'',
 25 					holdAction:function(){},
 26 					max:140 //默认最多140字
 27 				}
 28 				
 29 				$.extend(_cg,cg);
 30 				
 31 				for(var i in _cg){
 32 					this[i]=_cg[i];
 33 				}
 34 				
 35 				if(this.main=="" || this.wrap=="") return;
 36 				
 37 				var that=this,
 38 					main=$(that.main),
 39 					wrap=$(that.wrap),
 40 					holdTarget=$(that.holdTarget),
 41 					error=that.errorCls,
 42 					max=that.max;
 43 				var	holdfn=function(){
 44 					var l=parseInt(main.val().length),putout;
 45 					
 46 					if(l <= max || l == 0){
 47 						wrap.removeClass(error);
 48 						putout='你还可以输入<font>'+(max-l)+'</font>字';
 49 					}else{
 50 						wrap.addClass(error);
 51 						putout='已超出<font>'+(l-max)+'</font>字';
 52 					}
 53 					holdTarget.die('click');
 54 					
 55 					holdTarget.live('click',function(){
 56 						if(l<=max) that.holdAction();
 57 					});
 58 					
 59 					wrap.html(putout);
 60 				};
 61 					
 62 				holdfn();
 63 				main.keyup(holdfn);
 64 				main.focus(holdfn);
 65 				
 66 				this.update=holdfn;
 67 			}
 68 		};
 69 		
 70 		return fn;
 71 	};
 72 	
 73 	if(G && G.widget) G.widget.saycountdown=saycountdown;
 74 	
 75 })(window,GM);
