1 /**
  2  * @author fuqiang [designsor@gmail.com]
  3  * @date 20110726
  4  * 基于jquery1.6.2
  5  * Goodmedia前端js库core文件，负责创建GM对象和构建命名空间
  6  */
  7 (function(W,doc,$){
  8 	if(!W.GM) var GM={};
  9 	
 10 	//与业务紧密相关的-挂件
 11 	GM.widget={};
 12 	//与业务无关的比如overlay组件,最后扩展到jquery上，使用jquery的扩展机制进行最后的封装
 13 	GM.tools={};
 14 	//独立项目或者应用
 15 	GM.apps={};
 16 	
 17 	//是否是debug模式
 18 	GM.debug=function(){
 19 		var isdebug=(W.location.href.match('debug'));
 20 		return isdebug;
 21 	}();
 22 
 23 	//判断使用路径
 24 	//先找到当前的路径
 25 	GM.host=function(){
 26 		var scripts=doc.getElementsByTagName('script'),i,base;
 27 		for(i=0;i<scripts.length;i++){
 28 			var src=scripts[i].src,
 29 					namereg=/(GM-min|GM).js/;
 30 			if(namereg.test(src)){
 31 				base=src.slice(0,src.lastIndexOf('/')+1);
 32 			}
 33 		}
 34 		return base;
 35 	}();
 36 	
 37 	//转换到本地非压缩路径
 38 	GM.locality=function(uri){
 39 		return uri.replace('bulid','src').replace('-min','');
 40 	}
 41 	
 42 	GM.widget.host=GM.host + 'widget/';
 43 	GM.apps.host=GM.host + 'apps/';
 44 	
 45 	if(GM.debug){
 46 		GM.host='http://172.16.2.215/gm/bulid/';
 47 		GM.widget.host=GM.locality(GM.host) + 'widget/';
 48 		GM.apps.host=GM.locality(GM.host) + 'apps/';
 49 	}
 50 	
 51 	//额外加载项目文件 - 项目文件目前依赖关系依靠ant维护
 52 	GM.apps.require=function(appname,callback){
 53 		var appuri = GM.host + 'apps/'+appname+'/'+appname+'-min.js';
 54 		if(GM.debug) appuri=GM.locality(appuri);
 55 		$(function(){
 56 			$.getScript(appuri,function(){
 57 				if(callback) callback(GM.apps[appname]['exports']);
 58 			});
 59 		});
 60 	}
 61 	
 62 	//加载widget的方法
 63 	GM.widget.usemap={};
 64 	
 65 	GM.widget.use=function(widget,callback){
 66 		if(GM.widget.usemap.hasOwnProperty(widget)){
 67 			if(callback) callback(GM.widget);
 68 			return;
 69 		} 
 70 		var widgeturi = GM.host + 'widget/'+widget+'/'+widget+'-min.js';
 71 		if(GM.debug) widgeturi=GM.locality(widgeturi);
 72 		$(function(){
 73 			$.getScript(widgeturi,function(){
 74 				GM.widget.usemap[widget]=widgeturi;
 75 				if(callback) callback(GM.widget);
 76 			});
 77 		});
 78 	}
 79 	
 80 	//debug 模式下开启debug.js并且初始化debug面板
 81 	if(GM.debug){
 82 		GM.widget.use('debug',function(widget){
 83 				widget.debug.init();
 84 		});
 85 	} 
 86 	
 87 	W.GM=GM;
 88 })(window,document,jQuery);
