259 				function updateRt(active) {
260 					var s = $('.J_S').val(),
261 					is = $('.J_S').attr('checked'),
262 					city = $('.J_Citybtn').val() ? 'c=' + $('.J_Citybtn').val() : '',
263 					zone = $('.J_Zonebtn').val() ? '&z=' + $('.J_Zonebtn').val() : '',
264 					tag = $('.J_tagbtn').val() ? '&tag=' + $('.J_tagbtn').val() : '',
265 					project = $('.J_Projectbtn').val() ? '&s=' + $('.J_Projectbtn').val() : '';
266           if(project==='&s=热舞') project='&s=舞';
267 					if (project === '' && s) {
268 						if (s == '全部项目') s = '';
269 						project = '&s=' + s;
270 					} else if (project !== "" && s) {
271 						if (s == '全部项目') s = '';
272 						project = project + ' ' + s;
273 					}
274 					if (project == ('&s=' + s) && ! is) {
275 						project = '';
276 					}
277 					if (project == '&s=全部项目') project = '';
278 					if (zone == '&z=全部城区') zone = '';
279 					if (tag == '&tag=全部地标') tag = '';
280 					if (city == 'c=全部城市') city = '';
281 					var page = 1;
282 					if (typeof active == 'string' || typeof active == 'number') page = active;
283 					var q = $('#J_Q').val();
284 					if (q !== "请输入关键字") project = "&s=" + q;
285 					var arg = '?' + city + zone + tag + project + '&pageNo=' + page + '&max=10';
286 					//arg=encodeURIComponent(arg);
287 					arg = encodeURI(arg);
288 					if (lock) return;
289 					lock = true;
290 					if (!searchDate.hasOwnProperty(arg)) {
291 						$.ajax({
292 							url: _config["url"] + arg,
293 							success: function(data) {
294 								lock = false;
295                 var trimdata=data.replace(/\r\n/g,"");
296 								var ret = eval('(' + trimdata + ')');
297 								searchDate[arg] = ret;
298 								$('#J_siteResult').html(searchDate[arg]['Ret']);
299 								if (mypaging) mypaging.rebuild(searchDate[arg]['Count'], page);
300 								if (typeof active == 'function') active(ret);
301 							},
302 							error: function() {
303 								alert('网络错误，请重新选择一下看看');
304 								lock = false;
305 							}
306 						});
307 					} else {
308 						$('#J_siteResult').html(searchDate[arg]['Ret']);
309 						if (mypaging) mypaging.rebuild(searchDate[arg]['Count'], page);
310 						lock = false;
311 					}
312 					//window.location.href='/search/index.jsp'+arg;
313 				}
314 
315 			});
