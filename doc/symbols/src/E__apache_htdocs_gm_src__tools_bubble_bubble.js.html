1 /**
  2  * @author <a href="mailto:designsor@gmail.com" target="_blank">Fuqiang[designsor@gmail.com]</a>
  3  * @version 20110726
  4  * @fileoverview 气泡功能，自定义划过气泡，提供一些对外的接口和方法
  5  */
  6 (function(W, doc, $, G) {
  7 	/**
  8    * @memberOf jQuery
  9    * @constructor
 10    * @description 气泡浮出层管理，可以指定上下左右4个方向进行弹出
 11    */
 12 	var bubble = function(config) {
 13 		var _config = {
 14 			width: 0,
 15 			height: 0,
 16 			cls: '',
 17 			postion: 'bottom',
 18 			//left right top bottom 上下左右
 19 			target: '',
 20 			id: '#J_Bubble' + new Date().valueOf().toString().slice(3)
 21     };
 22 
 23 		$.extend(_config, config);
 24 
 25 		this.config = _config;
 26 	};
 27 
 28 	bubble.prototype = {
 29 		/**
 30      * @name jQuery.bubble#init
 31      * @description 初始化bubble功能
 32      */
 33 		init: function() {
 34 			var that = this,
 35 			cg = that.config;
 36 			that.createWrap(cg.target);
 37 		},
 38 		/**
 39      * @name jQuery.bubble#remove
 40      * @description 删除这个bubble
 41      */
 42 		remove: function() {
 43 			var that = this,
 44 			cg = that.config;
 45 			$(cg.id).remove();
 46 		},
 47 		/**
 48      * @name jQuery.bubble#show
 49      * @description 显示 
 50      */
 51 		show: function() {
 52 			var that = this,
 53 			cg = that.config;
 54 			$(cg.id).show();
 55 		},
 56 		/**
 57      * @name jQuery.bubble#hide
 58      * @description 隐藏
 59      */
 60 		hide: function() {
 61 			var that = this,
 62 			cg = that.config;
 63 			$(cg.id).hide();
 64 		},
 65 		/**
 66      * @name jQuery.bubble#setcontent
 67      * @description 设置气泡层内部html
 68      * @param {string} html 需要设置的内容
 69      */
 70 		setcontent: function(html) {
 71 			var that = this,
 72 			cg = that.config;
 73 			$(cg.id).html(html);
 74 		},
 75 		postionsign: function(target) {
 76 			var that = this,
 77 			cg = that.config,
 78 			postion = $(target).offset(),
 79 			boxH = $(target).height(),
 80 			boxW = $(target).width(),
 81 			_ret = {
 82 				bottom: {
 83 					left: postion.left + (boxW - cg.width) / 2,
 84 					top: postion.top + boxH
 85 				},
 86 				top: {
 87 					left: postion.left + (boxW - cg.width) / 2,
 88 					top: postion.top - cg.height
 89 				},
 90 				right: {
 91 					left: postion.left + boxW,
 92 					top: postion.top + (boxH - cg.height) / 2
 93 				},
 94 				left: {
 95 					left: postion.left - cg.width,
 96 					top: postion.top + (boxH - cg.height) / 2
 97 				}
 98 			};
 99 			return _ret;
100 		},
101     fixlocation:function(){
102       var that=this,cg=that.config;
103         var left=that.postionsign(cg.target)[cg.postion].left,
104             top=that.postionsign(cg.target)[cg.postion].top;
105             if(left<0) left=0; 
106             if(left>$('body').width()-cg.width) left=$('body').width()-cg.width; 
107       $(cg.id).offset({
108 					left:left,
109 					top:top
110 			});
111     },
112 		/**
113      * @name jQuery.bubble#createWrap
114      * @private
115      * @designsor 内部私有方法，构造整个bubble结构
116      */
117 		createWrap: function(target) {
118 			var that = this,
119 			cg = that.config;
120 
121 			//根据位置进行创建wrap
122 			if (that.postionsign(target).hasOwnProperty(cg.postion)) {
123 				var wrap = $('<div>').height(cg.height).width(cg.width).addClass(cg.cls).css({
124 					'z-index': 1200,
125 					'position': 'absolute',
126 					'display': 'none'
127 				}).attr('id', cg.id.slice(1));
128 				$('body').prepend(wrap);
129         that.fixlocation();
130 				var T;
131 				$(window).bind('scroll resize', function() {
132 					clearTimeout(T);
133 					setTimeout(function() {
134               that.fixlocation();
135 					},
136 				  30);
137 				});
138 
139 			} else {
140 				console.log('config.postion is error puts left,right,top or bottom');
141 			}
142 		}
143   };
144 
145 	$.extend({
146 		bubble: bubble
147 	});
148 
149 })(window, document, jQuery, GM);
