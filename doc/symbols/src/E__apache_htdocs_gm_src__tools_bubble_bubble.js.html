1 /**
  2  * @author <a href="mailto:designsor@gmail.com" target="_blank">Fuqiang[designsor@gmail.com]</a>
  3  * @version 20110726
  4  * @fileoverview 气泡功能，自定义划过气泡，提供一些对外的接口和方法
  5  */
  6 (function(W,doc,$,G){
  7   /**
  8    * @memberOf jQuery
  9    * @constructor
 10    * @description 气泡浮出层管理，可以指定上下左右4个方向进行弹出
 11    */
 12 	var bubble=function(config){
 13 		var _config={
 14 			width:0,
 15 			height:0,
 16 			cls:'',
 17 			postion:'bottom', //left right top bottom 上下左右
 18 			target:'',
 19 			id:'#J_Bubble'+new Date().valueOf().toString().slice(3)
 20 		}
 21 		
 22 		$.extend(_config,config);
 23 		
 24 		this.config=_config;
 25 	};
 26 	
 27 	
 28 	bubble.prototype={
 29     /**
 30      * @name jQuery.bubble#init
 31      * @description 初始化bubble功能
 32      */
 33 		init:function(){
 34 			var that=this,cg=that.config;
 35 			that.createWrap(cg.target);
 36 		},
 37     /**
 38      * @name jQuery.bubble#remove
 39      * @description 删除这个bubble
 40      */
 41 		remove:function(){
 42 			var that=this,cg=that.config;
 43 			$(cg.id).remove();
 44 		},
 45     /**
 46      * @name jQuery.bubble#show
 47      * @description 显示 
 48      */
 49 		show:function(){
 50 			var that=this,cg=that.config;
 51 			$(cg.id).show();
 52 		},
 53     /**
 54      * @name jQuery.bubble#hide
 55      * @description 隐藏
 56      */
 57 		hide:function(){
 58 			var that=this,cg=that.config;
 59 			$(cg.id).hide();
 60 		},
 61     /**
 62      * @name jQuery.bubble#setcontent
 63      * @description 设置气泡层内部html
 64      * @param {string} html 需要设置的内容
 65      */
 66 		setcontent:function(html){
 67 			var that=this,cg=that.config;
 68 			$(cg.id).html(html);
 69 		},
 70     /**
 71      * @name jQuery.bubble#createWrap
 72      * @private
 73      * @designsor 内部私有方法，构造整个bubble结构
 74      */
 75 		createWrap:function(target){
 76 			var cg=this.config,
 77 				postion=$(target).offset(),
 78 				boxH=$(target).height(),
 79 				boxW=$(target).width(),
 80 				postionsign={
 81 				bottom:{
 82 					left:postion.left-(cg.width/2),
 83 					top:postion.top+boxH
 84 				},
 85 				top:{
 86 					left:postion.left-(cg.width/2),
 87 					top:postion.top-cg.height
 88 				},
 89 				right:{
 90 					left:postion.left+boxW,
 91 					top:postion.top-(cg.width/2)
 92 				},
 93 				left:{
 94 					left:postion.left-cg.width,
 95 					top:postion.top-(cg.width/2)
 96 				}
 97 			};
 98 			
 99 			//根据位置进行创建wrap
100 			if(postionsign.hasOwnProperty(cg.postion)){
101 				var wrap=$('<div>')
102 					.height(cg.height)
103 					.width(cg.width)
104 					.addClass(cg.cls)
105 					.offset({
106 					left:postionsign[cg.postion].left,
107 					top:postionsign[cg.postion].top
108 					})
109 					.css({
110 						'z-index':50,
111 						'position':'absolute',
112 						'display':'none'
113 					})
114 					.attr('id',cg.id.slice(1));
115 				$('body').prepend(wrap);
116 			}else{
117 				console.log('config.postion is error puts left,right,top or bottom');
118 			}
119 		}
120 	}
121 	
122 	$.extend({
123 		bubble:bubble
124 	});
125 	
126 })(window,document,jQuery,GM);
