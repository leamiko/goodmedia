  7  */
  8 (function(W,G,$){
  9 	/**
 10 	 * @memberOf jQuery
 11 	 * @class
 12 	 * @description 存取改cookie值
 13 	 * @param {string} key cookie的键值
 14 	 * @param {string} value 设置cookie的新值
 15 	 * @param {objcet} options
 16 	 * @property {boolen} [raw=false] 是否encodeURIComponent
 17 	 * @property {number} [expires=-1] 过期时间，天为单位
 18 	 * @property {string} [path=""] cookie路径
 19 	 * @property {string} [domain=""] domain设置
 20 	 * @property {boolen} [secure="secure"] 默认是secure的
 21 	 * @example
 22 	 *  <h3>使用方法</h3>
 23 	 *	<p>Create session cookie:<br>
 24 	 *	$.cookie('the_cookie', 'the_value');<br>
 25 	 *	Create expiring cookie, 7 days from then:<br>
 26 	 *	$.cookie('the_cookie', 'the_value', { expires: 7 });<br>
 27 	 *	Create expiring cookie, valid across entire page:<br>
 28 	 *	$.cookie('the_cookie', 'the_value', { expires: 7, path: '/' });<br>
 29 	 *	Read cookie:<br>
 30 	 *	$.cookie('the_cookie'); // => 'the_value'<br>
 31 	 *	$.cookie('not_existing'); // => null<br>
 32 	 *	Delete cookie by passing null as value:<br>
 33 	 *	$.cookie('the_cookie', null);</p> 
 34 	 */
 35 	var cookie = function (key, value, options) {
 36 	    // key and at least value given, set cookie...
 37 	    if (arguments.length > 1 && String(value) !== "[object Object]") {
 38 	        options = jQuery.extend({}, options);
 39 	
 40 	        if (value === null || value === undefined) {
 41 	            options.expires = -1;
 42 	        }
 43 	
 44 	        if (typeof options.expires === 'number') {
 45 	            var days = options.expires, t = options.expires = new Date();
 46 	            t.setDate(t.getDate() + days);
 47 	        }
 48 	
 49 	        value = String(value);
 50 	
 51 	        return (document.cookie = [
 52 	            encodeURIComponent(key), '=',
 53 	            options.raw ? value : encodeURIComponent(value),
 54 	            options.expires ? '; expires=' + options.expires.toUTCString() : '', // use expires attribute, max-age is not supported by IE
 55 	            options.path ? '; path=' + options.path : '',
 56 	            options.domain ? '; domain=' + options.domain : '',
 57 	            options.secure ? '; secure' : ''
 58 	        ].join(''));
 59 	    }
 60 	
 61 	    // key and possibly options given, get cookie...
 62 	    options = value || {};
 63 	    var result, decode = options.raw ? function (s) { return s; } : decodeURIComponent;
 64 	    return (result = new RegExp('(?:^|; )' + encodeURIComponent(key) + '=([^;]*)').exec(document.cookie)) ? decode(result[1]) : null;
 65 	};
 66 	
 67 	$.extend({
 68 		cookie:cookie
 69 	});
 70 	
 71 })(window,GM,jQuery);
