  6 (function(W, $, G) {
  7 	var friends = function(uid, main,ids ,width, height) {
  8 		this.uid = uid;
  9 		this.main = main;
 10     this.ids=ids;
 11 		this.width = width || 315;
 12 		this.height = height || 150;
 13 		this.listId = '#J_Firends_' + now();
 14 	};
 15 
 16 	var Lock, T, timer;
 17 
 18 	function now() {
 19 		return new Date().valueOf();
 20 	}
 21 
 22 	friends.prototype = {
 23 		_timeout: function() {
 24       var that=this;
 25 			that.ready = true;
 26 			$(that.listId).append('<li data-id="-1">好友列表获取失败，请刷新重试</li>');
 27 			delete GM.apps.friends['_batchlist' + T];
 28 		},
 29 		_getlist: function() {
 30 			T = now();
 31 			var url = 'http://x.idongmi.com/api/followAjax.jsp',
 32 			that = this;
 33 			if (!Lock) {
 34 				Lock = true;
 35         GM.apps.friends['_batchlist'+T]=function(data){
 36           that._batchlist(data);
 37         };
 38 				$.getScript(url + '?uid=' + this.uid + '&callback=GM.apps.friends._batchlist' + T);
 39 				timer = setTimeout(function() {
 40             that._timeout();
 41 				},
 42 				5000);
 43 			}
 44 		},
 45 		_batchlist: function(data) {
 46 			var that = this;
 47 			Lock = false;
 48 			clearTimeout(timer);
 49 			//{
 50 			// s:[{
 51 			//  id:'xx',
 52 			//  name:'xx'
 53 			// },{
 54 			//  id:'xx',
 55 			//  name:'xx'
 56 			// }]
 57 			//}
 58 			$.each(data.s, function(index, o) {
 59 				$(that.listId).append('<li data-id="' + o.id + '">' + o.name + '</li>');
 60 			});
 61 			if (data.s.length === 0) {
 62 				$(that.listId).append('<li data-id="-1">你暂时还没有好友</li>');
 63 			}
 64 			that.ready = true;
 65       var height=that.height;
 66       if($(that.listId).height()>=height) $(that.listId).height(height); 
 67 			delete GM.apps.friends['_batchlist' + T];
 68 		},
 69 		hide: function() {
 70 			$(this.listId).hide();
 71 		},
 72 		show: function() {
 73 			$(this.listId).show();
 74 		},
 75 		_bulidlist: function(data) {
 76 			var that = this,
 77 			main = that.main,
 78 			height = that.height,
 79 			width = that.width,
 80 			id = that.listId.slice(1);
 81       $(main).after('<ul id="' + id + '" style="z-index:50;background:#fff;color:#a4a4a4;line-height:20px;display:none;overflow-y:auto;width:' + width + 'px;position:absolute;left:0px;top:' + ($(main).height()+5) + 'px;border:#c5c5c5 solid 1px;"></ul>');
 82 		},
 83 		init: function() {
 84       var that=this;
 85 			that._bulidlist();
 86       $(that.main).attr('autocomplete','off');
 87 			$(that.listId).find('li').live('mouseover', function() {
 88 				$(this).css({
 89 					background: '#e6f8fe',
 90 					color: '#3f9dfb'
 91 				});
 92 			}).live('mouseout', function() {
 93 				$(this).css({
 94 					background: '#fff',
 95 					color: '#a4a4a4'
 96 				});
 97       }).live('click',function(){
 98           var id=$(this).attr('data-id');
 99           if(id != -1){
100             var name=$(this).text(),
101                 val=$(that.main).val().replace('请从好友列表中选择昵称','');
102                 if(val.split(',').length>=5){
103                   alert('最多同时给五个人群发私信');
104                   that.hide();
105                   return;
106                 }
107                 var reg=new RegExp(name);
108                 if(!reg.test(val)){
109                   if(val===''){
110                     val=name;
111                   }else{
112                      val+=','+name;
113                   }
114                 }
115                 $(that.main).val(val);
116                 var ids=$(that.ids).val();
117                     idsarr=ids.split(',');
118                     if(idsarr[0]==="") idsarr.shift();
119                     idsarr.push(id);
120                     $(that.ids).val(idsarr.join(','));
121                 that.hide();
122           }
123       });
124       $(that.main).live('focus',function(){
125         that.hide();
126       });
127     $(that.main).live('keydown',function(e){
128         //阻止ctrl+v
129         if(e.ctrlKey && e.keyCode==86){
130           e.keyCode=0;
131           e.returnValue=false;
132           return false;
133         }
134         //修改对应的backspace和del按键
135         if(e.keyCode==8 || e.keyCode==46){
136           var myvalue=$(this).val(),
137               arr=myvalue.split(','),
138               str=arr.slice(0,arr.length-1).join(',');
139               if(str==',') str='';
140               $(this).val(str);
141 
142                var ids=$(that.ids).val(),
143                idsarr=ids.split(',');
144                idsarr.pop();
145                $(that.ids).val(idsarr.join(','));
146         }
147           return false;
148       });
149 			that._getlist();
150       $(that.ids).val("");
151 		}
152 	};
153 
154 	if (G && G.apps) {
155 		G.apps.friends = {
156 			exports: {
157 				friends: friends
158 			}
159 		};
160 	}
161 })(window, jQuery, GM);
