1 /**
  2  * @author <a href="mailto:designsor@gmail.com" target="_blank">Fuqiang[designsor@gmail.com]</a>
  3  * @version 20110726
  4  * @fileoverview 滚动木马组件，支持上下左右，自动延迟滚动
  5  */
  6 (function(W,doc,$,G){
  7 	/**
  8 	 * @memberOf jQuery
  9 	 * @constructor
 10 	 * @description 控制指定容器里的元素，上下左右滚动的组件
 11 	 * @param {objcet} cg
 12 	 * @property {element} wrap <p>#wrap外部包裹容器</p>
 13 	 * @property {element} wrapitem <p>.wrapitem内部包裹元素</p>
 14 	 * @property {string} [direction="left"] left|top方向
 15 	 * @property {number} [current=0] 当前指定滚动到第几页
 16 	 * @property {boolen} [auto=false] 是否自动滚动
 17 	 * @property {number} [autointerval=3000] 自动滚动间隔 毫秒
 18 	 */
 19 	var carousel=function(cg){
 20 		
 21 		var _carousel=function(){
 22 			/**
 23 			 * @private
 24 			 * @function
 25 			 * @param {object} cg 初始化carousel的参数
 26 			 * @description 根据config构建需要的dom结构
 27 			 */
 28 			var _bulid=function(cg){
 29 				var wrap=cg.wrap,wrapitem=cg.wrapitem;
 30 				if(wrap=="" || wrapitem=="") return;
 31 				
 32 				var firstChild=$(wrapitem).eq(0),w,h,size=$(wrapitem).size();
 33 				
 34 				if(cg.direction=='left'){
 35 					w=firstChild.outerWidth()*size;
 36 					h=firstChild.outerHeight();
 37 				}else if(cg.direction=='top'){
 38 					w=firstChild.outerWidth();
 39 					h=firstChild.outerHeight()*size;
 40 				}else{
 41 					return;
 42 				}
 43 				
 44 				var WrapDiv=$('<div>').css({
 45 					'height':h,
 46 					'width':w,
 47 					'position':'absolute'
 48 				}).addClass('fixclear');
 49 				
 50 				$(wrap).css({
 51 					'overflow':'hidden',
 52 					'position':'relative',
 53 					'width':firstChild.outerWidth(),
 54 					'height':firstChild.outerHeight()
 55 				}).prepend(WrapDiv);
 56 				
 57 				$(wrap).find(wrapitem).appendTo(WrapDiv);
 58 				
 59 				$(wrapitem).css('float','left');
 60 				
 61 				if($(wrap).find('.fixclear').length>1){
 62 					$(wrap).find('.fixclear').last().remove();
 63 				}
 64 				
 65 			};
 66 			
 67 			
 68 			return {
 69 				_init:function(o){
 70 					var _o={
 71 						wrap:'',
 72 						wrapitem:'',
 73 						direction:'left',
 74 						before:function(){},
 75 						after:function(){},
 76 						current:0,
 77 						endflg:true,
 78 						auto:false,
 79 						autointerval:3000
 80 					};
 81 
 82 					$.extend(_o,o);
 83 					
 84 					this.config=_o;
 85 					this.T=null;
 86 					
 87 					_bulid(this.config);
 88 					
 89 					if(this.config.auto){
 90 						this.auto();
 91 						this.autoEvent();
 92 					} 
 93 				}
 94 			}
 95 		}();
 96 		
 97 		_carousel._init.prototype={
 98 			/**
 99 			 * @name jQuery.carousel#forward
100 			 * @function
101 			 * @description 向前翻一页
102 			 */
103 			forward:function(){
104 				var that=this,config=that.config,
105 					l=$(config.wrap).find(config.wrapitem).length;
106 				if(config.endflg){
107 					config.endflg=false;
108 					config.current++;
109 					if(config.current>l-1) config.current=0;
110 					that.to(config.current);
111 				}
112 			},
113 			/**
114 			 * @name jQuery.carousel#backward
115 			 * @function
116 			 * @description 向后翻一页
117 			 */
118 			backward:function(){
119 				var that=this,config=that.config,
120 					l=$(config.wrap).find(config.wrapitem).length;
121 				if(config.endflg){
122 					config.endflg=false;
123 					config.current--;
124 					if(config.current<0) config.current=l-1;
125 					that.to(config.current);
126 				}
127 			},
128 			/**
129 			 * @name jQuery.carousel#to
130 			 * @function
131 			 * @description 翻到第几页
132 			 * @param {number} guide 页数指针
133 			 */
134 			to:function(guide){
135 				var that=this,config=that.config,moveObj={},
136 					Realwrap=$(config.wrap).find('.fixclear'),
137 					l=$(config.wrap).find(config.wrapitem).length;
138 				
139 				if(guide>l || guide<0) return;
140 				
141 				if(config.direction=='left'){
142 					moveObj={'left':'-'+$(config.wrap).width()*guide}
143 				}else if(config.direction=='top'){
144 					moveObj={'top':'-'+$(config.wrap).height()*guide}
145 				}
146 				that.before(guide);			
147 				Realwrap.animate(moveObj,500,function(){
148 					that.after(guide);
149 					config.endflg=true;
150 				});
151 			},
152 			/**
153 			 * @private
154 			 * @name jQuery.carousel-auto
155 			 * @function
156 			 * @description 激活自动滚动 |配置项没激活的话，这里可以唤醒
157 			 */
158 			auto:function(){
159 				var that=this,config=that.config;
160 				that.T=setInterval(function(){
161 					that.forward();
162 				},config.autointerval);
163 			},
164 			/**
165 			 * @private
166 			 * @function
167 			 * @name jQuery.carousel-autoEvent
168 			 * @description 绑定自动滚动需要的事件
169 			 */
170 			autoEvent:function(){
171 				var that=this,config=that.config;
172 				$(config.wrap).live('mouseenter',function(){
173 					clearInterval(that.T);
174 				}).live('mouseleave',function(){
175 					that.auto();
176 				});
177 			},
178 			/**
179 			 * @name jQuery.carousel#before
180 			 * @event
181 			 * @param {number} current 当前到第几页了
182 			 * @description 翻页之前触发
183 			 */
184 			before:function(current){
185 				var that=this,config=that.config;
186 				config.before(current);
187 			},
188 			/**
189 			 * @name jQuery.carousel#after
190 			 * @event
191 			 * @param {number} current 当前到第几页了
192 			 * @description 翻页之后触发
193 			 */
194 			after:function(current){
195 				var that=this,config=that.config;
196 				config.after(current);
197 			}
198 		};
199 		
200 		return new _carousel._init(cg);
201 	};
202 	
203 	$.extend({
204 		carousel:carousel
205 	});
206 	
207 })(window,document,jQuery,GM);
