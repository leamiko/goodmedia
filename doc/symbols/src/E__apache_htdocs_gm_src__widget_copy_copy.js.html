  6 (function(W,$,G){

  8    * @name GM.widget.copy

 12    * @param {String} txt 复制的文本
 13    * @param {Element} target 需要显示复制文本的区域
 14    */
 15 	var copy=function(btn,txt,target){
 16 		var _copy=function(txt){
 17 			   if($.browser.msie) {   
 18 	            	clipboardData.setData('Text',txt);

 24 	        			});

 27 	        			}
 28 	        		}else{
 29 		            	prompt("非IE内核浏览器，请复制以下地址：",txt); //去你妈的非IE内核...  
 30 	        		}   
 31 	        	}  
 32 		};
 33     $(btn).die();
 34 		$(btn).live('click',function(){
 35 			_copy(txt);
 36 		});
 37 };
 38 	
 39 	if(G && G.widget) G.widget.copy=copy;
