  6 (function(W, G, $) {
  7 	var backtop = function(target, config) {
  8 		var _config = {
  9 			effect: 'slow',
 10 			fixwindow: true,
 11 			right: 10,
 12 			bottom: 10
 13 		};
 14 		$.extend(_config, config);
 15 
 16 		var T;
 17 
 18 		if (_config.fixwindow) {
 19 			$(target).hide();
 20 			$(window).bind('scroll resize', function() {
 21 				function fixicon() {
 22 					var scrolltop = $(window).scrollTop(),
 23 					screenH = $(window).height(),
 24 					screenW = $(window).width(),
 25 					h = $(target).height(),
 26 					w = $(target).width();
 27 					if (scrolltop >= screenH) {
 28 						if ($.browser.msie && $.browser.version <= 6) {
 29 							$(target).css({
 30 								'position': 'absolute',
 31 								'left': screenW - _config.right - w,
 32 								'top': scrolltop - _config.bottom - h + screenH
 33 							});
 34 						} else {
 35 							$(target).css({
 36 								'position': 'fixed',
 37 								'right': _config.right,
 38 								'bottom': _config.bottom
 39 							});
 40 						}
 41 						$(target).fadeIn(_config.effect);
 42 					} else {
 43 						$(target).fadeOut();
 44 					}
 45 				}
 46 				if (T) clearTimeout(T);
 47 				T = setTimeout(fixicon, 100);
 48 			});
 49 		}
 50 
 51 		$(target).live('click', function() {
 52 			$(target).hide();
 53 			$(window).scrollTop(0);
 54 		});
 55 
 56 	};
 57 	if (G && G.widget) G.widget.backtop = backtop;
 58 })(window, GM, jQuery);
