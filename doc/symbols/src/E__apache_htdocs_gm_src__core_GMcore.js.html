1 /**
  2  * @author <a href="mailto:designsor@gmail.com" target="_blank">Fuqiang[designsor@gmail.com]</a>
  3  * @version 20110726
  4  * @fileoverview Goodmedia前端js库core文件，负责创建GM对象和构建命名空间 基于jquery1.6.2
  5  */
  6 (function(W,doc,$){
  7 	/**
  8 	 * @class
  9 	 * @name GM
 10 	 * @description 全局的GM对象
 11 	 */
 12 	
 13 	/**
 14 	 * @class
 15 	 * @name jQuery
 16 	 * @description 这里只对私自扩展的jQuery进行doc描述
 17 	 */
 18 	
 19 	if(!W.GM) var GM={};
 20 	/**
 21 	 * @namespace
 22 	 * @description 与业务紧密相关的-挂件
 23 	 */
 24 	GM.widget={};
 25 	/**
 26 	 * @namespace
 27 	 * @description 与业务无关的比如overlay组件,最后扩展到jquery上，使用jquery的扩展机制进行最后的封装
 28 	 */
 29 	GM.tools={};
 30 	/**
 31 	 * @namespace
 32 	 * @description 独立项目或者应用
 33 	 */
 34 	GM.apps={};
 35 	/**
 36 	 * @constant
 37 	 * @description 根据href里德debug关键字确定是否为debug模式
 38 	 */
 39 	GM.debug=function(){
 40 		var isdebug=(W.location.href.match('debug'));
 41 		return isdebug;
 42 	}();
 43 	/**
 44 	 * @constant
 45 	 * @description 获取当前js的根目录
 46 	 */
 47 	GM.host=function(){
 48 		var scripts=doc.getElementsByTagName('script'),i,base;
 49 		for(i=0;i<scripts.length;i++){
 50 			var src=scripts[i].src,
 51 					namereg=/(GM-min|GM).js/;
 52 			if(namereg.test(src)){
 53 				base=src.slice(0,src.lastIndexOf('/')+1);
 54 			}
 55 		}
 56 		return base;
 57 	}();
 58 	/**
 59 	 * @static
 60 	 * @description 转换到本地非压缩路径
 61 	 * @function
 62 	 * @param {string} uri
 63 	 * @returns {string} 处理过的路径
 64 	 */
 65 	GM.locality=function(uri){
 66 		return uri.replace(/bulid/g,'src').replace(/\-min/g,'');
 67 	}
 68 	/**
 69 	 * @constant
 70 	 * @description widget的根目录
 71 	 */
 72 	GM.widget.host=GM.host + 'widget/';
 73 	/**
 74 	 * @constant
 75 	 * @description apps的根目录
 76 	 */
 77 	GM.apps.host=GM.host + 'apps/';
 78 	
 79 	//debug模式下处理路径
 80 	if(GM.debug){
 81 		GM.host='http://172.16.2.215/gm/bulid/';
 82 		GM.widget.host=GM.locality(GM.host) + 'widget/';
 83 		GM.apps.host=GM.locality(GM.host) + 'apps/';
 84 	}
 85 	/**
 86 	 * @static
 87 	 * @description 额外加载项目文件 - 项目文件目前依赖关系依靠ant维护
 88 	 * @function
 89 	 * @param {string} appname
 90 	 * @param {function} callback
 91 	 */
 92 	GM.apps.require=function(appname,callback){
 93 		var appuri = GM.host + 'apps/'+appname+'/'+appname+'-min.js';
 94 		if(GM.debug) appuri=GM.locality(appuri);
 95 		$(function(){
 96 			$.getScript(appuri,function(){
 97 				if(callback) callback(GM.apps[appname]['exports']);
 98 			});
 99 		});
100 	}
101 	/**
102 	 * @static
103 	 * @description 把用过的widget储存，再次use则不再调用
104 	 * @private
105 	 */
106 	GM.widget.usemap={};
107 	/**
108 	 * @static
109 	 * @description 加载widget的方法
110 	 * @function
111 	 * @param {string} widget
112 	 * @param {function} callback
113 	 */
114 	GM.widget.use=function(widget,callback){
115 		if(GM.widget.usemap.hasOwnProperty(widget)){
116 			if(callback) callback(GM.widget);
117 			return;
118 		} 
119 		var widgeturi = GM.host + 'widget/'+widget+'/'+widget+'-min.js';
120 		if(GM.debug) widgeturi=GM.locality(widgeturi);
121 		$(function(){
122 			$.getScript(widgeturi,function(){
123 				GM.widget.usemap[widget]=widgeturi;
124 				if(callback) callback(GM.widget);
125 			});
126 		});
127 	}
128 	
129 	//debug 模式下开启debug.js并且初始化debug面板
130 	if(GM.debug){
131 		GM.widget.use('debug',function(widget){
132 				widget.debug.init();
133 		});
134 	}
135 	
136 	W.GM=GM;
137 })(window,document,jQuery);
