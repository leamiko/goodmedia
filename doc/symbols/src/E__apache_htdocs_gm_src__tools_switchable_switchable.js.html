  6 (function(W,G,$){
  7   /**
  8    * @memberOf jQuery

 10    */
 11 	var switchable=function(cg){
 12 		
 13 		function _switch(cg){
 14 			var _cg={
 15 				shownu:0, //默认第一个显示
 16 				targets:null, //elements
 17 				wraps:null, //elements
 18 				action:'click',
 19 				switchafter:function(){}
 20       };
 21 			
 22 			$.extend(_cg,cg);
 23 			
 24 			this.config=_cg;
 25 		}
 26 		
 27 		_switch.prototype={
 28       /**
 29        * @name jQuery.switchable#init
 30        * @description 初始化switch，参数均在config里设置
 31        */
 32 			init:function(){
 33 				var that=this,cg=that.config;
 34 				$(cg.wraps).hide();
 35 				$(cg.wraps).eq(cg.shownu).show();
 36 				$(cg.targets).live(cg.action,function(){
 37 					var ele=$(this),
 38 						index=ele.index(cg.targets),
 39 						wrap=$(cg.wraps).eq(index);
 40 						$(cg.wraps).hide();
 41 						wrap.show();
 42 						cg.shownu=index;
 43 						cg.switchafter(index,ele,wrap);
 44 				});
 45 			}
 46     };
 47 		
 48 		return new _switch(cg).init();
 49 		
 50 	};
 51 	
 52 	$.extend({
 53 		switchable:switchable
 54   });
 55
