1 /**
  2  * @author fuqiang[designsor@gmail.com]
  3  * @version 20110905
  4  * @fileoverview 哥实在扛不住了，所以哥决定写一个校验组件 -_-||
  5  */
  6 (function($,W,G){
  7 	/**
  8    * @name GM.widget.verify
  9    * @class
 10    * @description 表单校验组件，包含的校验规则存在rule中，自己查看源文件看吧
 11    * @param {Object} config 配置，详细看源文件……有默认的都写了。
 12    * @param {Object} rule [accessLevel] 如果有新增的校验或者独特的校验需求，在这里进行增加 
 13    */
 14 	var verify=function(config,rule){
 15 		
 16 		var _config={
 17 			form:null,
 18 			target:null, //form和target只能写一个
 19 			cls:null,
 20 			success:null,
 21 			blur:false,
 22 			batchcallback:null,
 23 			batchafter:null,
 24 			trim:true,
 25       checktrue:null,
 26 			focusfn:null,
 27 			attrname:'data-v' //data-v functionname:msg:arg.arg|functionname:msg:arg.arg
 28 		}
 29 		
 30 		var _rule={
 31 			empty:function(val){
 32 				if(val=="") return false;
 33 				return true;
 34 			},
 35 			num:function(val){
 36 				if(isNaN(val)) return false;
 37 				return true;
 38 			},
 39 			l:function(val,min,max){
 40 				if(val.length<min || val.length>max) return false;
 41 				return true;
 42 			},
 43 			n:function(val,min,max){
 44 				if(val<min || val>max) return false;
 45 				return true;
 46 			},
 47 			phone:function(val){
 48 				return (/^(?:13\d|15\d|18\d)-?\d{5}(\d{3}|\*{3})$/).test(val)
 49 			},
 50 			email:function(val){
 51 				return (/^([a-zA-Z0-9_-])+@([a-zA-Z0-9_-])+((\.[a-zA-Z0-9_-]{2,3}){1,2})$/).test(val);
 52 			},
 53 			equal:function(val,selected){
 54 				if(val!=$(selected).val()) return false;
 55 				return true;
 56 			},
 57 			checked:function(val,node){
 58 				if(!$(node).attr('checked')) return false;
 59 				return true;
 60 			},
 61 			sp:function(val){
 62 				var reg=/[\~|\!|\@|\#|\$|\%|\^|\&|\*|\(|\)|\,|\s|\\|\||>|<|\?|\/|\;|\:|\"|\'|\{|\}|\[|\]|\`]/g;
 63 				if(val.match(reg)) return false;
 64 				return true;
 65 			}
 66 		}
 67 		
 68 		$.extend(_config,config);
 69 		$.extend(_rule,rule);
 70 		
 71 		this.rule=_rule;
 72 		this.config=_config;
 73 		this.data={};
 74 	}
 75 	
 76 	verify.prototype={
 77     /**
 78      * @name GM.widget.verify#init
 79      * @description 目前对外的结构就只有一个init方法，所有回调和自定义事件都在cfg里体现
 80      */
 81 		init:function(){
 82 			var that=this,cg=that.config;
 83 			
 84 			if(!cg.cls) return; //cls 钩子必须写
 85 			
 86 			if((!cg.form && !cg.target) || (cg.form && cg.target)) return; //只能存在一个..
 87 			
 88 			if(cg.form){
 89 				that._form(cg.form)
 90 			}else if(cg.target){
 91 				that._target(cg.target)
 92 			}
 93 			
 94 			if(cg.blur){
 95 				$(cg.cls).live('blur',function(){
 96 						var node=$(this);
 97 						that._bacthone(node);
 98 				});
 99 				$(cg.cls).live('focus',function(){
100 						var node=$(this);
101 						if(cg.focusfn) cg.focusfn(node);
102 				})
103 			}
104 		},
105 		_form:function(form){
106 			var that=this,cg=that.config;
107 			$(form).live('submit',function(){
108 				var flg=that._batch(cg.cls);
109 				if(flg && cg.success) cg.success(that.data);
110 				return false;
111 			});
112 		},
113 		_target:function(btn){
114 			var that=this,cg=that.config;
115 			$(btn).live('click',function(){
116 				var flg=that._batch(cg.cls);
117 				if(flg && cg.success) cg.success(that.data);
118 				return false;
119 			});
120 		},
121 		_translate:function(str){
122 			
123 			//str=functionname,msg,arg.arg|functionname,msg,arg.arg
124 			
125 			var fnary=str.split('|'),returnobj={};
126 			
127 			for(var i=0;i<fnary.length;i++){
128 				var thisfnary=fnary[i].split(':'),
129 					name=thisfnary[0],
130 					msg=thisfnary[1],
131 					argary=(thisfnary[2]) ? thisfnary[2].split('.') : [];
132 				returnobj[name]={
133 					msg:msg,
134 					arg:argary
135 				};
136 			}
137 			
138 			return returnobj;
139 			
140 		},
141 		_bacthone:function(node){
142 			var that=this,cg=that.config,
143 					val=$.trim(node.val()),
144 					vstr=node.attr(cg.attrname),
145 					rulevalobj=that._translate(vstr);
146           if(cg.trim) $(node).val(val);
147 					for(var i in rulevalobj){
148 						rulevalobj[i]['arg'].splice(0,0,val);
149 						var examine=that.rule[i].apply(this,rulevalobj[i]['arg']);
150 						if(!examine){
151 							if(cg.batchcallback) cg.batchcallback(val,rulevalobj[i]['msg'],node);
152 							break;
153 						}else{
154               if(cg.checktrue) cg.checktrue(node);
155             }
156 					}
157 		},
158 		_batch:function(cls){
159 			var that=this,cg=that.config,flg=true,j=0;
160 			
161 			$(cls).each(function(index,node){
162 				var val=$.trim($(this).val()),
163 					vstr=$(this).attr(cg.attrname),
164 					rulevalobj=that._translate(vstr);
165           if(cg.trim) $(node).val(val);
166 					j++; //用做data标示，有name用name，没name用j
167 					/*
168 					rulevalobj={
169 						rule:msg
170 					}
171 					*/
172 				for(var i in rulevalobj){
173 					rulevalobj[i]['arg'].splice(0,0,val);
174 					var examine=that.rule[i].apply(this,rulevalobj[i]['arg']);
175 					if(!examine){
176 						if(cg.batchcallback) cg.batchcallback(val,rulevalobj[i]['msg'],node);
177 						flg=false;
178 						break;
179 					}else{
180             var node=$(this);
181             if(cg.checktrue) cg.checktrue(node);
182           }
183 				}
184 				
185         if(!flg){
186 					return false; //中断each循环
187 				}else{
188 					if($(this).attr('name')){
189 						that.data[$(this).attr('name')]=val;
190 					}else{
191 						that.data[j]=val;
192 					}
193 				} 
194 			});
195 			return flg;
196 		}
197 	}
198 	
199 	if(G && G.widget) G.widget.verify=verify;
200 	
201 })(jQuery,window,GM);
