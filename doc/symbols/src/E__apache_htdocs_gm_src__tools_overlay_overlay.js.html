1 /**
  2  * @author <a href="mailto:designsor@gmail.com" target="_blank">Fuqiang[designsor@gmail.com]</a>
  3  * @version 20110726
  4  * @fileoverview 通用浮出层功能部分,最后扩展成为jquery插件形式被调用
  5  */
  6 (function(W,doc,$,G){
  7 	/**
  8    * @memberOf jQuery
  9    * @constructor
 10 	 * @description 对IE6进行兼容shade fixed
 11 	 */
 12 	var overlay=function(cg){
 13 
 14 		var _overlay=function(){
 15 			
 16 			return {
 17 				_init:function(o){
 18 					/**
 19 					* 默认配置
 20 					* cover 默认有遮罩  50%的灰色，不可自己定义, 可以用css后期自己改，统一所有对象使用一个#J_GM_Cover
 21 					* drag 默认不可以拖拽，先不实现，后期再实现 20110726 -涉及到多层互相覆盖的问题，比如页面上有多个overlay，这个是目前不允许的
 22 					* content 内容为HTML，直接进行注入
 23 					*/
 24 					var _o={
 25 						wrapId:'J_GM_OverlayWrap', //默认的全局wrap-ID
 26 						wrapCls:'J_GM_OverlayWrapCls',
 27 						coverId:'J_GM_Cover',
 28 						width:null,
 29 						height:null,
 30 						content:'',
 31 						cover:true,
 32 						drag:false,
 33 						opacity:0.5
 34 					};
 35 
 36 					$.extend(_o,o);
 37 
 38 					this.config=_o;
 39 
 40 				}
 41 			}
 42 			
 43 		}();
 44 
 45 		/**
 46 		 * 开放出去的通用api
 47 		 */
 48 		_overlay._init.prototype={
 49       /**
 50        * @name jQuery.overlay#fire
 51        * @param {String} html [accessLevel] 覆盖掉覆盖层内部区域的字符串
 52        * @param {Function} callback [accessLevel] 触发浮出层后的回调
 53        * @description 触发覆盖层浮出
 54        */
 55 			fire:function(html,callback){
 56 				var that=this,config=that.config;
 57 
 58 				if(config._destroy) return;
 59 
 60 				if(!doc.getElementById(config.wrapId)){
 61 
 62 					var wrap=$('<div>',{id:config.wrapId});
 63 
 64 					wrap.appendTo('body').html(config.content);
 65 
 66 					if(html) $('#'+config.wrapId).html(html);
 67 					
 68 					//取得wrap的高宽直接传入reset
 69 					//that.reset(config.width,config.height,config.wrapCls);
 70 					
 71 					if(config.cover && $('#'+config.coverId).length==0) that._cover();
 72 
 73 				}else{
 74 					if(html) $('#'+config.wrapId).html(html);
 75 
 76 					$('#'+config.coverId).show();
 77 
 78 					$('#'+config.wrapId).show();
 79 				}
 80 				
 81 				$('#'+config.wrapId).css('display','inline');
 82 				
 83 				//高度的获取有时候会延迟在IE6下
 84 				setTimeout(function(){
 85 					that.reset($('#'+config.wrapId).innerWidth(),$('#'+config.wrapId).innerHeight(),config.wrapCls);
 86 				},10);
 87 				
 88 				if(callback) callback(wrap);
 89 
 90 			},
 91       /**
 92        * @name jQuery.overlay#opacity
 93        * @param {Number} opacity 透明度值 
 94        * @description 可以重新设置透明
 95        */
 96 			opacity:function(opacity){
 97 				var that=this,config=that.config;
 98 				config.opacity=opacity;
 99 			},
100       /**
101        * @name jquery.overlay#reset
102        * @param {Number} w 宽度
103        * @param {Number} h 高度
104        * @param {Number} cls 样式名
105        * @description 重写设置高宽和样式
106        */
107 			reset:function(w,h,cls){
108 
109 				var that=this,config=that.config;
110 
111 				config.width=w;
112 				config.height=h;
113 				config.wrapCls=cls;
114 				
115 
116 				$('#'+config.wrapId).css({
117 					'width':w,
118 					'height':h,
119 					'z-index':1000
120 				}).addClass(cls);
121 				
122 				that._fix($('#'+config.wrapId),that);
123 				
124 			},
125       /**
126        * @name jQuery.overlay#close
127        * @description 关闭覆盖层
128        * @param {Function} callback 关闭后的回调
129        */
130 			close:function(callback){
131 
132 				var that=this,config=that.config;
133 
134 				$('#'+config.coverId).hide();
135 
136 				$('#'+config.wrapId).hide();
137 				
138 				$('#'+config.wrapId).css({
139 					'width':'auto',
140 					'height':'auto'
141 				});
142 
143 				if(callback) callback($('#'+config.coverId),$('#'+config.wrapId));
144 
145 			},
146 			//遮罩+拖拽把手
147 			_cover:function(){
148 
149 				var that=this,config=that.config,
150 					cover=$('<div>',{id:config.coverId}),
151 					ie6=($.browser.msie && $.browser.version=='6.0');
152 
153 				cover.css({
154 					'background-color':'#ccc',
155 					'opacity':config.opacity,
156 					'height': doc.documentElement.clientHeight,
157 					'z-index':999
158 				}).prependTo('body');
159 
160 		        if (ie6) {
161 		            cover.append("<iframe style='width:100%;" +
162 		                "filter:alpha(opacity=0);" +
163 		                "z-index:-1;'>");
164 		            cover.find('iframe').css('height',cover.height());
165 		        }
166 		        
167 		        that._fixScroll();
168 
169 			},
170 			//fix复位
171 			_fix:function(node,host){
172 
173 				var that=host,config=that.config,
174 					ie6=($.browser.msie && $.browser.version=='6.0'),
175 					scrollTop=$(doc).scrollTop();
176 					
177 				node.css({
178 					'position':ie6 ? 'absolute' : 'fixed',
179 					'zoom':1,
180 					'left':'50%',
181 					'top':ie6 ? ((doc.documentElement.offsetHeight-config.height)/2)+scrollTop : '50%',
182 					'margin-left':-(config.width/2),
183 					'margin-top':ie6 ? 0 : -(config.height/2)
184 				});
185 				
186 				$('#'+config.coverId).css({
187 					'position':ie6 ? 'absolute' : 'fixed',
188 		            'left':0,
189 		            'top':ie6 ? scrollTop : 0,
190 		            'width':ie6 ?  Math.max(doc.documentElement.clientWidth,doc.body.clientWidth) : "100%"
191 		        });
192 				
193 			},
194 			_fixScroll:function(){
195 				var that=this,config=that.config;
196 				$(W).bind('scroll resize',{host:that},that._windowBind);
197 			},
198 			_windowBind:function(e){
199 				var host=e.data.host;
200 				host._fix($('#'+host.config.wrapId),host);
201 				$('#'+host.config.coverId).css({'height': doc.documentElement.clientHeight});
202 			},
203       /**
204        * @name jQuery.overlay#destroy
205        * @description 注销覆盖层
206        */
207 			destroy:function(){
208 				var that=this;config=that.config;
209 				$('#'+config.coverId).remove();
210 				$('#'+config.wrapId).remove();
211 				$(W).unbind('scroll resize',that._windowBind);
212 				if(G.tools.overlay) G.tools.overlay=null;
213 			}
214 		};
215 
216 		/**
217 		 * 每次引用都返回唯一一个全局的浮层对象
218 		 */
219 
220 		if(!G.tools.overlay) G.tools.overlay=new _overlay._init(cg);
221 
222 		return 	G.tools.overlay;
223 	};
224 
225 	//扩展到jquery原型上，不绑定到GM上
226 	$.extend({
227 		overlay:overlay
228 	});
229 
230 })(window,document,jQuery,GM);
